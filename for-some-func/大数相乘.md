# **大数相乘**



```cpp
#define max 100 
string mul(string s1,string s2)//大于0的两位数乘法
{
	if(s1=="0" || s2 =="0") return "0";
	string ans;
    int a[max]={0},b[max]={0},c[max+1]={0},i=0,j=0,m,n;
    m=s1.size()-1;
    n=s2.size()-1;
    for (i=0;m>=0;m--)//将字符串1转换为数字，并反序存入
        a[i++]=s1[m]-'0';
    for (j=0;n>=0;n--)//将字符串2转换为数字，并反序存入
        b[j++]=s2[n]-'0';
    ////////////////////////////////////////////////////
    //"123"*"4567"变成[321] 
    ////////////////////////////////////////////////////
    for (i=0;i<s1.size();i++)//运算存入
    {
        for (j=0;j<s2.size();j++)
            c[i+j]+=a[i]*b[j];//重点
    }
    for (i=0;i<=max;i++)
    {
        c[i+1]=c[i+1]+c[i]/10;
        c[i]=c[i]%10;
    }
    for (i=max;i>=0 && c[i]==0;i--);
    for (;i>=0;i--)
    {
        ans += c[i] + '0';
    }
    return ans;
}
```
